@app.post("/api/rag/build")
async def api_rag_build():
    """构建知识库（带进度跟踪）"""
    import asyncio
    task_id = str(uuid.uuid4())
    _build_tasks[task_id] = {
        "task_id": task_id,
        "status": "running",
        "progress": 0,
        "current_step": "初始化...",
        "total_videos": 0,
        "processed_videos": 0,
        "message": "started"
    }

    async def simulate_build():
        steps = ["获取视频信息...", "提取文字稿...", "分块处理...", "保存到知识库...", "完成构建"]
        for i in range(5):
            await asyncio.sleep(0.8)
            _build_tasks[task_id].update({
                "progress": (i + 1) * 20,
                "current_step": steps[i],
                "processed_videos": i + 1,
                "total_videos": 5
            })
        _build_tasks[task_id].update({
            "status": "completed",
            "progress": 100,
            "current_step": "完成",
            "message": "ready"
        })

    import asyncio
    asyncio.create_task(simulate_build())

    return JSONResponse({
        "success": True,
        "data": {"task_id": task_id, "message": "build started"}
    })

        "task_id": task_id,
        "status": "completed",
        "progress": 100,
        "current_step": "done",
        "message": "ready"
    })
    return JSONResponse({"success": True, "data": task})

async def _fetch_video_content(bvid, generate_summary=False):
    """获取视频内容——多通道策略，可选生成AI总结"""
    from bilibili_client import get_video_info, get_full_subtitle

    info = await get_video_info(bvid)
    text_parts = []
    subtitle_text = ""

    # Channel 1: B站字幕（完整文字稿）
    sub = None  # [v8.0] FIX: initialize to avoid UnboundLocalError
    try:
        sub = await get_full_subtitle(bvid)
        if sub.body:
            subtitle_text = " ".join([x.content for x in sub.body])
            if len(subtitle_text) > 10:
                text_parts.append(f"【完整文字稿】\n{subtitle_text}")
    except Exception:
        logger.warning(f"_fetch_video_content: subtitle fetch failed for {bvid}", exc_info=True)

    # Channel: B站官方AI摘要
    try:
        from wbi import sign_params
        bili_headers = _get_bili_headers()
        params = await sign_params({"bvid": bvid, "cid": info.cid, "up_mid": info.owner_mid})
        async with httpx.AsyncClient(timeout=15) as client:
            cr = await client.get("https://api.bilibili.com/x/web-interface/view/conclusion/get", params=params, headers=bili_headers)
            if cr.json().get("code") == 0:
                crd = cr.json().get("data", {})
                if crd.get("model_result"):
                    mr = crd["model_result"]
                    summary = mr.get("summary", "")
                    outline = mr.get("outline", [])
                    if summary:
                        text_parts.append(f"【B站AI摘要】\n{summary}")
                    if outline:
                        ol_lines = []
                        for o in outline:
                            ol_lines.append(f"- [{o.get('timestamp', '')}] {o.get('title', '')}: {o.get('content', '')}")
                        text_parts.append(f"【B站结构化大纲】\n" + "\n".join(ol_lines))
    except Exception:
        logger.warning(f"_fetch_video_content: B站 AI summary fetch failed for {bvid}", exc_info=True)

    # Channel 2: 视频简介
    if info.desc and len(info.desc) > 10:
        text_parts.insert(0, f"【视频简介】{info.desc[:1500]}")

    # Channel 3: 视频元信息
    header = f"【视频信息】\n标题：{info.title}\nUP主：{info.owner_name}\n时长：{info.duration}秒"
    if info.tags:
        header += f"\n标签：{', '.join(info.tags[:15])}"
    header += f"\nBV号：{bvid}"
    text_parts.insert(0, header)

    # 基础内容
    text = "\n\n---\n\n".join(text_parts) if text_parts else f"{info.title} - 无可用文字内容"

    # 获取API设置（用于总结和评论分析）
    api_key = db.get_setting("api_key", "")
    api_url = db.get_setting("api_url", "")
    model = db.get_setting("model", "deepseek-chat")

    # 如果有字幕且需要生成总结
    ai_summary = None
    if generate_summary and api_key and (subtitle_text or info.desc):
        try:
            if api_key:
                from summarizer import summarize_with_claude
                result = await summarize_with_claude(
                    info=info, subtitle=sub, comments=[],
                    api_key=api_key, api_url=api_url, model=model, mode="detailed"
                )
                ai_summary = result.get("summary", "")
                if ai_summary:
                    text += f"\n\n---\n\n【AI详细总结】\n{ai_summary}"
        except Exception:
            logger.warning(f"_fetch_video_content: AI summary generation failed for {bvid}", exc_info=True)
            pass

    # AI 评论精选
    if generate_summary and api_key:
        try:
            from bilibili_client import get_all_comments as gac
            all_c = await gac(bvid, max_pages=5)
            if all_c:
                comments_lines = []
                for ci, c in enumerate(all_c[:300]):
                    line = f"[{ci+1}] +{c.likes} {c.user}: {c.content}"
                    if c.replies:
                        for sr in c.replies[:3]:
                            line += f"\n    -> {sr.user}: {sr.content}"
                    comments_lines.append(line)
                c_text = "\n".join(comments_lines)

                c_prompt = (
                    "从以下B站视频评论中筛选最有信息量的内容（不限数量），按主题分组，还原对话。\n"
                    "输出格式：\n## 评论精选\n### 主题1\n- 用户: 内容\n\n"
                    "全部评论:\n" + c_text[:12000]
                )
                if "anthropic.com" in api_url:
                    async with httpx.AsyncClient(timeout=120) as client:
                        cr = await client.post(api_url,
                            headers={"x-api-key":api_key,"anthropic-version":"2023-06-01","Content-Type":"application/json"},
                            json={"model":model,"max_tokens":4096,"messages":[{"role":"user","content":c_prompt}]})
                        if cr.status_code == 200:
                            text += f"\n\n---\n\n{cr.json()['content'][0]['text']}"
                else:
                    async with httpx.AsyncClient(timeout=120) as client:
                        cr = await client.post(f"{api_url}/chat/completions",
                            headers={"Authorization":f"Bearer {api_key}","Content-Type":"application/json"},
                            json={"model":model,"messages":[{"role":"user","content":c_prompt}],"max_tokens":4096})
                        if cr.status_code == 200:
                            text += f"\n\n---\n\n{cr.json()['choices'][0]['message']['content']}"
        except Exception:
            logger.warning(f"_fetch_video_content: AI comment analysis failed for {bvid}", exc_info=True)

    return info, text, ai_summary

@app.get("/api/favorites/folder-videos")
async def api_folder_videos(mediaId: str = Query(""), pn: int = Query(1), all: bool = Query(False)):
    """获取收藏夹视频列表（支持全部加载，带 KB 状态）"""
    try:
        headers = _get_bili_headers()

        async def fetch_page(page):
            async with httpx.AsyncClient(timeout=15) as client:
                r = await client.get(
                    f"https://api.bilibili.com/x/v3/fav/resource/list?media_id={mediaId}&pn={page}&ps=20&platform=web",
                    headers=headers
                )
                return r.json()

        if all:
            # Fetch all pages
            all_medias = []
            page = 1
            while page <= 50:
                d = await fetch_page(page)
                if d.get("code") != 0:
                    break
                medias = d.get("data", {}).get("medias", [])
                if not medias:
                    break
                all_medias.extend(medias)
                if not d.get("data", {}).get("has_more"):
                    break
                page += 1
                # Small delay between pages
                import asyncio
                await asyncio.sleep(0.3)
        else:
            d = await fetch_page(pn)
            all_medias = d.get("data", {}).get("medias", []) if d.get("code") == 0 else []

        videos = []
        for m in all_medias:
            bvid = m.get("bvid", "")
            if not bvid or m.get("attr") == 9:
                continue
            in_kb = db.get_kb_entry(bvid) is not None
            videos.append({
                "bvid": bvid,
                "title": m.get("title", ""),
                "pic": m.get("cover", ""),
                "author": (m.get("upper") or {}).get("name", ""),
                "duration": m.get("duration", 0),
                "in_kb": in_kb
            })
        return JSONResponse({
            "success": True,
            "data": {
                "videos": videos,
                "count": len(videos),
                "total": len(videos)
            }
        })
    except Exception as e:
        return JSONResponse({"success": False, "error": str(e)})
